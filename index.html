  <script>

    function formatAmount(value) {

      return "₹" + Math.round(value).toLocaleString("en-IN");

    }


    function calculateCashback() {

      var inputValue = document.getElementById("amount").value;

      var amount = Number(inputValue.replace(/,/g, "").trim());


      var errorBox = document.getElementById("errorBox");

      var resultBox = document.getElementById("resultBox");


      if (!amount || amount <= 0) {

        errorBox.style.display = "block";

        resultBox.style.display = "none";

        return;

      }


      errorBox.style.display = "none";


      var slab1 = 0;

      var slab2 = 0;

      var slab3 = 0;


      if (amount <= 5000) {

        slab1 = amount * 0.02;

      } else if (amount <= 40000) {

        slab1 = 5000 * 0.02;

        slab2 = (amount - 5000) * 0.05;

      } else {

        slab1 = 5000 * 0.02;

        slab2 = 35000 * 0.05;

        slab3 = (amount - 40000) * 0.07;

      }


      var total = slab1 + slab2 + slab3;


      if (total > 4000) {

        total = 4000;

      }


      document.getElementById("spendAmount").innerText = formatAmount(amount);

      document.getElementById("slab1").innerText = formatAmount(slab1);

      document.getElementById("slab2").innerText = formatAmount(slab2);

      document.getElementById("slab3").innerText = formatAmount(slab3);

      document.getElementById("totalCashback").innerText = formatAmount(total);


      resultBox.style.display = "block";

    }


    document.getElementById("amount").addEventListener("keyup", function(event) {

      if (event.key === "Enter") {

        calculateCashback();

      }

    });
